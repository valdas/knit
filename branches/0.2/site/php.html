        function printArray($nums, $max)

        function swp($nums, $count)
        {
                $hold=$nums[$count];
                $nums[$count] = $nums[$count+1];
                $nums[$count+1] = $hold;
                return $nums;
        }


        function sortArray($nums, $max)
        {
                $swap=1;


                while ($swap==1)
                {
                        $count=0;


                        $swap=0;


                        while ($count<$max)
                        {
                                if($nums[$count]>$nums[$count+1])
                                {
                                        $nums = swp($nums, $count);
                                        $swap=1;
                                }
                $count++;
                        }


                }
        return $nums;


        }
