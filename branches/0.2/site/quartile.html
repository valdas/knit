 function quartile(data, opt) {
     s2a("shrink,2,offC,_,medianC,|,onC,*,f,5.1f,truncate,0.05,outer,1",
         opt,",")
     quartile1(data,opt)
 } 
 function quartile1(data,opt,  f,i,s,s1,n,min,q1,median,q3,max) {
     n  = asort(data,s)
     if (opt["truncate"]) {
         lower = within(round(n*   opt["truncate"]),1,n)
         upper = within(round(n*(1-opt["truncate"])),1,n)
         n     = 0
         for(i=lower;i<=upper;i++)
             s1[++n]= s[i]
     } else 
         for(i in s) s1[i]=s[i]
     min   = s1[1];
     q1    = s1[round(1*n/4)];
     median= s1[round(2*n/4)];
     q3    = s1[round(3*n/4)];
     max   = s1[n];
     f=opt["f"]
     printf("%"f",%"f",%"f",%"f",%"f",[%s]\n",
            min,q1,median,q3,max,
            quartile2(min,q1,median,q3,max,opt))
 }
